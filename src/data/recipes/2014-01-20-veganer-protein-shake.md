  prepTime: "PT3M"

  servings: "2 Portionen"
  ingredients:
    - "500g Sojajoghurt Natur"
    - "200g TK-Himbeeren"
    - "200ml Wasser oder Sojamilch"
    - "1 Handvoll Mandeln"
    - "Etwas Vanille bzw. Mark ½ Vanilleschote"
    - "Mit Agavendicksaft oder Xylitol nach Belieben süßen"
  instructions:
